Clean Structure:
    dotnet new sln -n UserService
    dotnet new classlib -n UserService.Core -f net9.0
    dotnet new classlib -n UserService.Infrastructure -f net9.0
    dotnet new webapi -n UserService.API -f net9.0
    dotnet sln add UserService.Core
    dotnet sln add UserService.Infrastructure
    dotnet sln add UserService.API
    dotnet add UserService.API reference UserService.Core
    dotnet add UserService.API reference UserService.Infrastructure
    dotnet add UserService.Infrastructure reference UserService.Core

    UserService.API =>
        Controllers/UserController.cs

    UserService.Core =>
        Entities => Models => User.cs
        Entities => Interfaces => IUserService.cs

    UserService.Infrastructure =>
        Repositories => UserRepository.cs
        Services => UserService.cs

    Program.cs

    cd UserService.API
    dotnet run
